# Este programa recibe a y b, e imprime los primos en [a, b].
R a1
R b1
i1 a1
  L 2
  n0 i1
  C check_prime
  I == p0 1 3
  L 4
  i1 + i1 1
  I <= i1 b1 2
E
L 3
  O i1
  G 4
# Funcion que calcula a % b.
# Parámetro 1: a0
# Parámetro 2: b0
# Retorna: a0 % b0, guardado en c0
F mod
  c0 / a0 b0
  c0 * c0 b0
  c0 - a0 c0
  RET mod
# Funcion de calcula si un numero es primo.
# Parámetro 1: n0
# Retorna: 1 si n0 es primo, guardado en p0.
# Utiliza variables: i0, a0, b0, z0, p0
# Utiliza labels: 8080, 8081, 8082
F check_prime
  I <= n0 1 8081
  I == n0 2 8082
  i0 1
  L 8080
  i0 + i0 1
  a0 n0
  b0 i0
  C mod
  I == c0 0 8081
  z0 * i0 i0
  I <= z0 n0 8080
  G 8082
  L 8081
  p0 0
  RET check_prime
  L 8082
  p0 1
  RET check_prime
