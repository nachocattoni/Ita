# Programa que lee enteros n, e imprime 1 si son primos, 0 si no.
# Esta vez el programa usa funciones.
L 1
R n0
C check_prime
O p0
G 1
# Funcion que calcula a % b.
# Parámetro 1: a0
# Parámetro 2: b0
# Retorna: a0 % b0, guardado en c0
F mod
  c0 / a0 b0
  c0 * c0 b0
  c0 - a0 c0
  RET mod
# Funcion de calcula si un numero es primo.
# Parámetro 1: n0
# Retorna: 1 si n0 es primo, guardado en p0.
# Utiliza variables: i0, a0, b0, z0
# Utiliza labels: 8080, 8081, 8082
F check_prime
  I <= n0 1 8081
  I == n0 2 8082
  i0 1
  L 8080
  i0 + i0 1
  a0 n0
  b0 i0
  C mod
  I == c0 0 8081
  z0 * i0 i0
  I <= z0 n0 8080
  G 8082
  L 8081
  p0 0
  RET check_prime
  L 8082
  p0 1
  RET check_prime
